using System;

namespace Phase0_Exercise46_Numeric_Die

{

public class Die

{

int sides;

public Die(int s)

{

if (s < 1)

throw new ArgumentException("Number of sides must be greater than one", "sides");

else

sides = s;

}

public int Sides()

{

return sides;

}

public int Roll()

{

var random = new Random();

int roll = random.Next(0, sides);

return roll+1;

}

}

}
